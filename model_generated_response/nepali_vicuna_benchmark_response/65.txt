    """
    Binary search algorithm to find a specific element in a sorted array
    :param arr: the sorted array
    :param key: the element to search for
    :return: the index of the element in the array, or -1 if the element is not found
    """
    low = 0
    high = len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == key:
            return mid
        elif arr[mid] < key:
            low = mid + 1
        else:
            high = mid - 1
    return -1
